### 3.1 CaseManager — 案件管理器

**// gdscript
**## 案件管理器：管理案件数据、案件切换、连锁伏笔
class_name CaseManager
extends Node

var current_case: CaseData
var completed_cases: PackedStringArray = []
var foreshadowing_state: Dictionary = {}

## 加载并启动案件
func start_case(case_id: String) -> void:
    current_case = load("res://data/cases/%s.tres" % case_id)
    CaseEventBus.case_started.emit(case_id)
    CaseEventBus.case_region_loaded.emit(case_id, current_case.region.related_tiles.size())

## 完成当前案件
func complete_case(final_score: int) -> void:
    var case_id = current_case.case_id
    completed_cases.append(case_id)
    _check_foreshadowing(case_id)
    CaseEventBus.case_completed.emit(case_id, final_score)

## 伏笔检查：检测当前案件完成后是否触发下一案件的伏笔
func _check_foreshadowing(completed_case_id: String) -> void:
    for link in current_case.foreshadowing_links:
        if _are_conditions_met(link.conditions):
            foreshadowing_state[link.target_case_id] = {
                "source": completed_case_id,
                "hint": link.hint_text,
                "unlocked": true
            }

## 获取案件库中可复玩的案件列表
func get_replayable_cases() -> Array[CaseData]:
    var cases: Array[CaseData] = []
    for case_id in completed_cases:
        cases.append(load("res://data/cases/%s.tres" % case_id))
    return cases

### 3.2 SceneController — 场景控制器

**// gdscript
**## 场景控制器：管理当前场景的呈现、交互和跳转
class_name SceneController
extends Node2D

@export var scene_image: TextureRect
@export var hotspot_container: Node2D
@export var npc_container: Node2D

var current_scene: SceneData
var exploration_rate: float = 0.0

## 进入场景
func enter_scene(scene_id: String) -> void:
    current_scene = load("res://data/scenes/%s.tres" % scene_id)
    scene_image.texture = current_scene.texture
    _setup_hotspots(current_scene.hotspots)
    _setup_npcs(current_scene.npcs)
    exploration_rate = 0.0
    SceneEventBus.scene_entered.emit(scene_id)

## 离开场景
func exit_scene() -> void:
    SceneEventBus.scene_exited.emit(current_scene.id)

## 热区点击处理
func _on_hotspot_clicked(hotspot: HotspotData) -> void:
    match hotspot.type:
        "CLUE":
            ClueSystem.discover_clue(hotspot.clue_id, "observation")
        "ITEM":
            _show_item_detail(hotspot)
        "NAV":
            enter_scene(hotspot.target_scene_id)
    _update_exploration_rate()

## 更新探索率，>80%标记为"已探索"
func _update_exploration_rate() -> void:
    var total = current_scene.hotspots.size()
    var clicked = _count_clicked_hotspots()
    exploration_rate = float(clicked) / float(total) if total > 0 else 1.0
    if exploration_rate >= 0.8:
        SceneEventBus.scene_completed.emit(current_scene.id)

### 3.3 DialogueSystem — 对话系统

→ 对话系统规范参见《00_标准化开发目录》v2.19 B-7《NPC与对话系统》/ B-8《对话树设计规范》
**v4.0新特性**：对话后线索筛选、对话变体池、知识边界、四层情感参数
**MVP范围**：对话后线索筛选 + 四层情感参数为MVP必做；对话变体池为M2扩展；知识边界MVP简化实现（基础字段校验）
**// gdscript
**## 对话系统：管理对话树遍历、选项渲染、表情/势态反馈、对话后线索筛选
## v4.0特性：对话变体池、知识边界校验、四层情感参数动态影响选项
class_name DialogueSystem
extends Node

var current_dialogue: DialogueData
var current_node: DialogueNodeData
var is_active: bool = false

## 对话中收集的待筛选线索 {clue_id: clue_data}
var pending_clues: Dictionary = {}

## 开始对话
func start_dialogue(dialogue_id: String) -> void:
    current_dialogue = load("res://data/dialogues/%s.tres" % dialogue_id)
    # 普通模式：从变体池中随机选择一组变体（M2+扩展，MVP暂用默认组）
    if DifficultyManager.current_mode == DifficultyManager.Mode.NORMAL:
        current_dialogue.select_variant()
    current_node = current_dialogue.get_start_node()
    is_active = true
    pending_clues.clear()
    DialogueEventBus.dialogue_started.emit(current_dialogue.npc_id)
    _render_node(current_node)

## 选择对话选项
func make_choice(choice_index: int) -> void:
    if not is_active or current_node == null:
        return
    var choice = current_node.choices[choice_index]

    DialogueEventBus.dialogue_choice_made.emit(choice.choice_id, current_node.id)

    # 知识边界校验：NPC不会回答超出其知识范围的问题
    if not _check_knowledge_boundary(choice):
        _render_ignorance_response()
        return

    # 收集对话中发现的线索（进入待筛选池，对话后统一处理）
    for clue_id in choice.clues_triggered:
        var clue_data = load("res://data/clues/%s.tres" % clue_id)
        pending_clues[clue_id] = clue_data

    # 更新NPC四层情感参数（影响后续对话选项）
    _update_emotion_params(choice)

    # 更新NPC表情
    if choice.expression_change != "":
        DialogueEventBus.expression_changed.emit(current_dialogue.npc_id, choice.expression_change)

    # 更新洞察之星（对话选择影响调查方向质量）
    if choice.insight_impact != "none":
        for chain_id in choice.affected_chains:
            StarRatingSystem.update_insight_stars(chain_id, choice.insight_impact, 0.0)

    # 跳转下一节点
    if choice.next_node != "":
        current_node = current_dialogue.get_node(choice.next_node)
        _render_node(current_node)
    else:
        end_dialogue()

## 结束对话（对话型场景Step 2：进入线索筛选界面）
func end_dialogue() -> void:
    is_active = false
    DialogueEventBus.dialogue_ended.emit(current_dialogue.npc_id)

    # 对话型场景Step 2适配：如果有待筛选线索，进入线索筛选界面
    if pending_clues.size() > 0:
        _open_clue_screening()

## 线索筛选：玩家勾选认为重要的线索（对话型场景Step 2→Step 3）
func confirm_clue_screening(selected_clue_ids: Array) -> void:
    # 玩家勾选的线索正式进入推理墙
    for clue_id in selected_clue_ids:
        if pending_clues.has(clue_id):
            ClueSystem.discover_clue(clue_id, "dialogue_screened")

    # 未勾选的线索也会被收集（不扣分，全收集原则），但标记为"未主动识别"
    for clue_id in pending_clues.keys():
        if clue_id not in selected_clue_ids:
            ClueSystem.discover_clue(clue_id, "dialogue_passive")
            # 影响观察之星（漏识别的线索不计入"主动发现"统计）
            StarRatingSystem.mark_clue_passively_found(clue_id)

    pending_clues.clear()
    DialogueEventBus.clue_screening_completed.emit(selected_clue_ids.size())

## 渲染当前对话节点（难度+情感双重过滤选项）
func _render_node(node: DialogueNodeData) -> void:
    var visible_choices = DifficultyManager.filter_choices(node.choices)
    # 四层情感参数过滤：信任度过低时隐藏敏感选项，紧张度过高时关闭追问选项
    visible_choices = _filter_by_emotion(visible_choices)
    UIManager.show_dialogue(node, visible_choices)

## 打开线索筛选界面（对话型场景Step 2）
func _open_clue_screening() -> void:
    var clue_list = []
    for clue_id in pending_clues.keys():
        clue_list.append(pending_clues[clue_id])
    UIManager.show_clue_screening(clue_list)

    var emotions = npc_data.emotion_params
    match choice.emotion_impact:
        "trust_up":     emotions["trust"] = min(emotions["trust"] + 0.15, 1.0)
        "trust_down":   emotions["trust"] = max(emotions["trust"] - 0.2, 0.0)
        "tension_up":   emotions["tension"] = min(emotions["tension"] + 0.2, 1.0)
        "tension_down": emotions["tension"] = max(emotions["tension"] - 0.1, 0.0)
        "hidden_up":    emotions["hidden"] = min(emotions["hidden"] + 0.15, 1.0)
    # 稳定性越低，情绪波动越大（乘以stability系数）
    npc_data.emotion_params = emotions

## 情感参数过滤：根据当前情感状态过滤可选对话选项

    var emotions = npc_data.emotion_params
    var result = []
    for choice in choices:
        # 信任度<0.3时隐藏需要深度信任的选项
        if choice.requires_trust and emotions["trust"] < 0.3:
            continue
        # 紧张度>0.7时关闭追问类选项
        if choice.is_pressure_option and emotions["tension"] > 0.7:
            continue
        result.append(choice)
    return result

func _calculate_verification_level(evidence: Array) -> String:
    # MVP简化：支持证据数-反驳证据数判定档位
    var support = 0
    var refute = 0
    for e in evidence:
        if e.type == "support": support += 1
        elif e.type == "refute": refute += 1
    if refute > 0: return "CONTRADICTORY"
    if support >= 3: return "VERIFIED"
    if support >= 1: return "SUPPORTED"
    return "INSUFFICIENT"

        if tree.has_node(hypothesis_id):
            return tree
    return null

    return a + "|" + b if a < b else b + "|" + a

**// gdscript
**## 难度管理器：管理三种难度模式的状态与概率判定
## 核心设计：难度不影响数据层，只影响"系统替玩家做了多少事"

class_name DifficultyManager
extends Node

enum Mode { EASY, NORMAL, HARD }

var current_mode: Mode = Mode.NORMAL
var hint_base_probability: float = 0.7    # 普通模式基础提示概率（与02文档一致：70%）
var hint_current_probability: float = 0.7
var hint_max_probability: float = 0.95    # 普通模式概率上限（动态调整天花板）
var _interactions_since_last_clue: int = 0
var _stall_interaction_threshold: int = 15  # 连续15次交互未发现线索视为停滞
var _wrong_inference_count: int = 0       # 连续错误推理次数

## 设置难度
func set_mode(mode: Mode) -> void:
    current_mode = mode
    match mode:
        Mode.EASY:
            hint_current_probability = 1.0
        Mode.NORMAL:
            hint_current_probability = hint_base_probability
        Mode.HARD:
            hint_current_probability = 0.0   # 困难模式被动提示为0，仅保留主动求助

## 概率判定：普通模式核心机制
func roll_hint_probability() -> bool:
    match current_mode:
        Mode.EASY:   return true
        Mode.HARD:   return false   # 困难模式无被动提示
        Mode.NORMAL: return randf() < hint_current_probability
    return false

## 误导线索出现概率
func get_mislead_probability() -> float:
    match current_mode:
        Mode.EASY:   return 0.0     # 简单模式无误导
        Mode.NORMAL: return 0.3     # 普通模式30%概率
        Mode.HARD:   return 0.7     # 困难模式70%概率
    return 0.0

## 进度检测与概率调整（普通模式动态调整机制）
## 调整规则：停滞+10% / 错误推理+5% / 正常进展-5%
func on_progress_check() -> void:
    if current_mode != Mode.NORMAL: return
    if _interactions_since_last_clue >= _stall_interaction_threshold:
        hint_current_probability = minf(hint_max_probability, hint_current_probability + 0.1)
        UIEventBus.hint_probability_adjusted.emit(hint_current_probability, "stall_detected")
    elif _wrong_inference_count >= 3:
        hint_current_probability = minf(hint_max_probability, hint_current_probability + 0.05)
        UIEventBus.hint_probability_adjusted.emit(hint_current_probability, "wrong_inference")
    else:
        hint_current_probability = maxf(hint_base_probability, hint_current_probability - 0.05)

## 记录交互（由各系统在玩家操作时调用）
func record_interaction() -> void:
    _interactions_since_last_clue += 1

## 重置停滞计数（线索发现时由ClueSystem调用）
func reset_stall_counter() -> void:
    _interactions_since_last_clue = 0

## 记录错误推理
func record_wrong_inference() -> void:
    _wrong_inference_count += 1

## 重置错误计数（推理正确时调用）
func reset_wrong_inference() -> void:
    _wrong_inference_count = 0

## 过滤对话选项（困难模式隐藏推理提示选项）

## 设计原则：不扣分、错误可回溯重做，星级仅反映最终提交状态
## 单条推理链满星 9⭐（观察3⭐ + 推理3⭐ + 洞察3⭐）
## 全案件14条推理链，总满星 126⭐
class_name StarRatingSystem
extends Node

## 每条推理链的星级评定 {chain_id: ChainStarRating}
var chain_ratings: Dictionary = {}

## ── 单条推理链星级结构 ──
class ChainStarRating:
    var chain_id: String
    var observation_stars: int = 0   # 观察之星 0-3⭐
    var reasoning_stars: int = 0     # 推理之星 0-3⭐
    var insight_stars: int = 0       # 洞察之星 0-3⭐
    var first_try_perfect: bool = false  # 首次提交即全对（神枪手徽章依据）

    func get_total_stars() -> int:
        return observation_stars + reasoning_stars + insight_stars

    func get_grade() -> String:
        var total = get_total_stars()
        if total >= 9: return "gold"
        if total >= 6: return "silver"
        if total >= 3: return "bronze"
        return "none"

## ── 观察之星：线索收集完整度 ──
## 判定规则：纯按缺失条数离散档位（与线索重要性等级无关）
## 缺 ≥3 条 = 1⭐ | 缺 1-2 条 = 2⭐ | 全齐 = 3⭐
func update_observation_stars(chain_id: String, total_clues: int, found_clues: int) -> void:
    var missing = total_clues - found_clues
    var stars: int
    if missing <= 0:
        stars = 3
    elif missing <= 2:
        stars = 2
    else:
        stars = 1
    _ensure_chain(chain_id).observation_stars = stars
    UIEventBus.star_rating_updated.emit(chain_id, "observation", stars)

## ── 推理之星：最终提交正确比例 ──
## 判定规则：只看最终提交的正确判断比例，试错过程不扣分
## 正确比例 < 50% = 1⭐ | 50%-89% = 2⭐ | ≥90% = 3⭐
func update_reasoning_stars(chain_id: String, correct_ratio: float, is_first_submission: bool) -> void:
    var stars: int
    if correct_ratio >= 0.9:
        stars = 3
    elif correct_ratio >= 0.5:
        stars = 2
    else:
        stars = 1
    var chain = _ensure_chain(chain_id)
    chain.reasoning_stars = stars
    if is_first_submission and correct_ratio >= 1.0:
        chain.first_try_perfect = true
    UIEventBus.star_rating_updated.emit(chain_id, "reasoning", stars)

## ── 洞察之星：调查方向与隐藏信息发现 ──
## 判定规则：方向选择质量 + 隐藏细节发现
## 走关键方向+发现全部隐藏细节 = 3⭐ | 走一般方向或部分隐藏 = 2⭐ | 走误导方向 = 1⭐
func update_insight_stars(chain_id: String, direction_quality: String, hidden_found_ratio: float) -> void:
    var stars: int
    match direction_quality:
        "key":
            stars = 3 if hidden_found_ratio >= 0.9 else 2
        "normal":
            stars = 2
        "misleading":
            stars = 1
        _:
            stars = 1
    _ensure_chain(chain_id).insight_stars = stars
    UIEventBus.star_rating_updated.emit(chain_id, "insight", stars)

## 获取单条推理链星级
func get_chain_stars(chain_id: String) -> Dictionary:
    if not chain_ratings.has(chain_id):
        return {"observation": 0, "reasoning": 0, "insight": 0, "total": 0}
    var cr = chain_ratings[chain_id] as ChainStarRating
    return {
        "observation": cr.observation_stars,
        "reasoning": cr.reasoning_stars,
        "insight": cr.insight_stars,
        "total": cr.get_total_stars(),
        "grade": cr.get_grade()
    }

## 获取案件总星级
func get_case_total_stars() -> Dictionary:
    var obs_sum = 0
    var rea_sum = 0
    var ins_sum = 0
    var total_chains = 0
    var perfect_first_try_count = 0
    for cr in chain_ratings.values():
        var c = cr as ChainStarRating
        obs_sum += c.observation_stars
        rea_sum += c.reasoning_stars
        ins_sum += c.insight_stars
        total_chains += 1
        if c.first_try_perfect:
            perfect_first_try_count += 1
    var total = obs_sum + rea_sum + ins_sum
    var max_stars = total_chains * 9
    var ratio = float(total) / float(max_stars) if max_stars > 0 else 0.0
    return {
        "observation_total": obs_sum,
        "reasoning_total": rea_sum,
        "insight_total": ins_sum,
        "total_stars": total,
        "max_stars": max_stars,
        "ratio": ratio,
        "perfect_first_try_chains": perfect_first_try_count
    }

func _ensure_chain(chain_id: String) -> ChainStarRating:
    if not chain_ratings.has(chain_id):
        var cr = ChainStarRating.new()
        cr.chain_id = chain_id
        chain_ratings[chain_id] = cr
    return chain_ratings[chain_id] as ChainStarRating

**// gdscript
**## 调查进度系统：事件驱动的进度推进、固定/灵活进度事件、软锁与绕过去机制
## 设计原则：进度是剧情自然推进的结果，不采用硬性时间计量；
## 时间作为沉浸感表现层存在（维多利亚时代时钟氛围），但不做倒计时惩罚；

class_name ProgressSystem
extends Node

## 当前进度点（事件驱动，整数递增，每个事件推进1点）
var current_progress: int = 0
## 当前案件的进度总点数（用于显示进度百分比）
var total_progress_points: int = 0
## 沉浸用显示时间（游戏内小时制，随进度推进，仅用于氛围营造）
var display_time: float = 9.0
var display_date: String = "1881-03-05"

## 固定进度事件 {progress_point: [ProgressEvent]}
var fixed_events: Dictionary = {}

## 推进进度（由案件事件触发，不是按真实时间流逝）
func advance_progress(points: int, source_event: String) -> void:
    var from = current_progress
    current_progress += points
    _check_fixed_events()
    # 同步推进沉浸显示时间（每进度点对应约30分钟游戏内时间，仅氛围用）
    display_time += float(points) * 0.5
    if display_time >= 24.0:
        display_time -= 24.0
        _advance_date()
    MapEventBus.progress_advanced.emit(from, current_progress, source_event)
    DifficultyManager.on_progress_check()

## 添加固定进度事件（到达某进度点时触发）
func add_fixed_event(event: ProgressEvent) -> void:
    if not fixed_events.has(event.trigger_progress):
        fixed_events[event.trigger_progress] = []
    fixed_events[event.trigger_progress].append(event)

## 检查并触发到达进度点的事件
func _check_fixed_events() -> void:
    var to_trigger: Array[ProgressEvent] = []
    for prog in fixed_events.keys():
        if prog <= current_progress:
            for ev in fixed_events[prog]:
                to_trigger.append(ev)
            fixed_events.erase(prog)
    # 按进度点排序后依次触发
    to_trigger.sort_custom(func(a, b): return a.trigger_progress < b.trigger_progress)
    for ev in to_trigger:
        MapEventBus.fixed_progress_event_triggered.emit(ev.id, ev.event_type)

## 时间锁检测：检查某事件是否被进度锁阻挡
## 返回：{is_locked: bool, lock_type: "soft"/"hard", bypass_methods: [], remedy_methods: [], failure_msg: ""}
func check_progress_lock(event_id: String) -> Dictionary:
    var event_data = CaseManager.get_event_data(event_id)
    if event_data == null or not event_data.has("lock_type"):
        return {"is_locked": false}
    if event_data.lock_type == "none":
        return {"is_locked": false}
    # 检查是否满足解锁条件
    var required_progress = event_data.get("required_progress", -1)
    if required_progress >= 0 and current_progress >= required_progress:
        return {"is_locked": false}
    return {
        "is_locked": true,
        "lock_type": event_data.lock_type,
        "lock_target": event_data.get("lock_target", ""),
        "bypass_methods": event_data.get("bypass_methods", []),
        "remedy_methods": event_data.get("remedy_methods", []),
        "failure_msg": event_data.get("bypass_failure_msg", "")
    }

## 格式化沉浸显示时间字符串（仅氛围用，不作为核心机制）
func get_display_time_string() -> String:
    var hour = int(display_time)
    var minute = int((display_time - hour) * 60)
    var period = "上午" if hour < 12 else "下午"
    var display_hour = hour if hour <= 12 else hour - 12
    return "%s%d:%02d" % [period, display_hour, minute]

## 获取当前进度百分比
func get_progress_percentage() -> float:
    if total_progress_points <= 0: return 0.0
    return float(current_progress) / float(total_progress_points)

class ProgressEvent:
    var id: String
    var trigger_progress: int   # 触发的进度点
    var event_type: String      # "scene_unlock" / "npc_appear" / "clue_available" / "cutscene"
    var description: String

var current_ending: EndingTier = EndingTier.PROBATION

## 根据案件总星级占比判定结局
func determine_ending() -> EndingTier:
    var case_stats = StarRatingSystem.get_case_total_stars()
    var ratio = case_stats.ratio

    if ratio >= 0.9:
        current_ending = EndingTier.LEGENDARY
    elif ratio >= 0.7:
        current_ending = EndingTier.OUTSTANDING
    elif ratio >= 0.5:

## 构建倒排索引（加速关键词检索）
func _build_search_index() -> void:
    for domain_id in knowledge_data:
        var domain = knowledge_data[domain_id]
        for subtopic_id in domain.subtopics:
            var subtopic = domain.subtopics[subtopic_id]
            for entry_id in subtopic.entries:
                var entry = subtopic.entries[entry_id]
                # 标题词索引（权重最高）
                for word in entry.title.split(" "):
                    _add_to_index(word, entry_id, 3)
                # 正文关键词索引
                for keyword in entry.keywords:
                    _add_to_index(keyword, entry_id, 2)
                # 正文提及索引（权重最低）
                for word in entry.content.substr(0, 200).split(" "):
                    if word.length() > 2:
                        _add_to_index(word, entry_id, 1)

func _add_to_index(word: String, entry_id: String, weight: int) -> void:
    word = word.to_lower()
    if not search_index.has(word):
        search_index[word] = {}
    if not search_index[word].has(entry_id):
        search_index[word][entry_id] = 0
    search_index[word][entry_id] += weight

## ── 关键词检索（4种使用方式之一） ──
## 简单模式：智能联想（模糊匹配+相关推荐）
## 普通模式：标准检索（精确+模糊）
## 困难模式：仅精确匹配 + 矛盾信息干扰
func search(query: String) -> Array:
    var results = []
    var query_words = query.to_lower().split(" ")

    # 根据难度调整检索策略
    var mode = DifficultyManager.current_mode

    for word in query_words:
        if search_index.has(word):
            for entry_id in search_index[word]:
                var weight = search_index[word][entry_id]
                _add_result(results, entry_id, weight)

    # 简单模式：补充智能联想推荐
    if mode == DifficultyManager.Mode.EASY:
        results = _expand_with_smart_suggestion(results, query_words)

    # 困难模式：仅保留精确匹配（移除模糊匹配结果）
    if mode == DifficultyManager.Mode.HARD:
        results = _filter_exact_match(results, query_words)
        # 困难模式随机插入10%矛盾信息干扰（M2+，MVP暂不启用）
        pass

    # 按相关度排序
    results.sort_custom(func(a, b): return a.score > b.score)

    return results

## ── 交叉引用（4种使用方式之三，M2+扩展） ──
func get_related_entries(entry_id: String) -> Array:
    # MVP简化：按共享关键词数量推荐相关条目
    var entry = _find_entry(entry_id)
    if entry == null: return []

    var related = {}
    for keyword in entry.keywords:
        if search_index.has(keyword):
            for other_id in search_index[keyword]:
                if other_id != entry_id:
                    if not related.has(other_id):
                        related[other_id] = 0
                    related[other_id] += search_index[keyword][other_id]

    var result = []
    for rid in related:
        result.append({"entry_id": rid, "score": related[rid]})
    result.sort_custom(func(a, b): return a.score > b.score)
    return result.slice(0, 5)  # 最多推荐5条

    if all_entries.size() == 0: return {}

func flush_pending_sync() -> void:
    if pending_sync_queue.size() == 0: return
    # 只同步最新的快照（队列中最后一条）
    var latest = pending_sync_queue[-1]
    pending_sync_queue.clear()
    APIManager.get_instance().sync_save(latest)

## 加载游戏（带冲突合并策略）
func load_game() -> bool:
    if AuthManager.is_registered():
        var server_data = await APIManager.get_instance().fetch_save()
        if server_data != null:
            var local_data = _load_local()
            if local_data.size() > 0:
                # 冲突合并：比较时间戳，取较新的
                var local_ts = local_data.get("timestamp", 0.0)
                var server_ts = server_data.get("timestamp", 0.0)
                if local_ts > server_ts:
                    _apply_save_data(local_data)
                    # 本地更新，需同步到服务器
                    APIManager.get_instance().sync_save(local_data)
                else:
                    _apply_save_data(server_data)
            else:
                _apply_save_data(server_data)
            return true
    var local_data = _load_local()
    if local_data != null:
        _apply_save_data(local_data)
        return true
    return false
