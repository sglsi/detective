📖 Godot 使用示例

# 在 Godot 中使用像素艺术表情

# 1. 导入资源
# 将 PNG 文件放入：res://assets/portraits/pixel/

# 2. 在对话系统中加载表情
func _load_pixel_expressions():
    var expression_map = {}
    var expressions = [
        "喜悦", "开心", "兴奋", "自信", "坚定",
        "疑惑", "沉默", "思考", "凝思", "疲惫",
        "生气", "愤怒", "神秘", "狡黠"
    ]
    
    for expr in expressions:
        var path = "res://assets/portraits/pixel/sherlock_%s.png" % expr
        if ResourceLoader.exists(path):
            expression_map[expr] = load(path)
    
    return expression_map

# 3. 在对话渲染器中使用
func _update_expression(mood: String):
    if mood in expression_map:
        $PortraitSprite.texture = expression_map[mood]
